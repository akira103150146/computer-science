📝 T05　組合數 C(n,k)（用函式） 進階

寫一個 comb(n,k) 函式回傳組合數 C(n,k) = n! / (k!(n−k)!)。請用「邊乘邊除」避免階乘溢位（C(n,k) = C(n,k−1)×(n−k+1)/k 的迭代寫法）。

輸入 輸出

10 3120
看參考解

C 參考解

#include <stdio.h>
long long comb(int n, int k){
    if (k < 0 || k > n) return 0;
    if (k > n - k) k = n - k;
    long long r = 1;
    for (int i = 0; i < k; i++) r = r * (n - i) / (i + 1);
    return r;
}
int main(void){ int n, k; scanf("%d %d", &n, &k); printf("%lld\n", comb(n, k)); return 0; }

Python 參考解

def comb(n, k):
    if k < 0 or k > n:
        return 0
    k = min(k, n - k)
    r = 1
    for i in range(k):
        r = r * (n - i) // (i + 1)
    return r
n, k = map(int, input().split())
print(comb(n, k))

#include <stdio.h>
#include <string.h>
int oddSum(const char *s){
    int n = (int)strlen(s), sum = 0;
    for (int i = 0; i < n; i++) if ((n - 1 - i) % 2 == 0) sum += s[i] - '0';  /* 由右數第1,3,5... */
    return sum;
}
int evenSum(const char *s){
    int n = (int)strlen(s), sum = 0;
    for (int i = 0; i < n; i++) if ((n - 1 - i) % 2 == 1) sum += s[i] - '0';
    return sum;
}

    char s[16]; scanf("%15s", s);
    int d = oddSum(s) - evenSum(s);
    printf("%d\n", d < 0 ? -d : d);
    return 0;
}

Python 參考解

def odd_sum(s):
    return sum(int(c) for i, c in enumerate(s[::-1]) if i % 2 == 0)
def even_sum(s):
    return sum(int(c) for i, c in enumerate(s[::-1]) if i % 2 == 1)
s = input().strip()
print(abs(odd_sum(s) - even_sum(s)))

C 參考解

#include <stdio.h>
#include <string.h>
void toBase(long long n, int base, char *out){
    const char *D = "0123456789ABCDEF";
    char buf[64]; int k = 0;
    if (n == 0){ out[0] = '0'; out[1] = '\0'; return; }
    while (n > 0){ buf[k++] = D[n % base]; n /= base; }
    for (int i = 0; i < k; i++) out[i] = buf[k - 1 - i];
    out[k] = '\0';
}
int main(void){
    long long n; int base; scanf("%lld %d", &n, &base);
    char out[64]; toBase(n, base, out);
    printf("%s\n", out);
    return 0;
}

Python 參考解

def to_base(n, base):
    if n == 0:
        return "0"
    digs = "0123456789ABCDEF"
    out = ""
    while n > 0:
        out = digs[n % base] + out
        n //= base
    return out
n, base = map(int, input().split())
print(to_base(n, base))

#include <stdio.h>
int a[25];
void sortAsc(int *x, int n){            /* 用傳址修改陣列：示範參數傳址 */
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - 1 - i; j++)
            if (x[j] > x[j + 1]){ int t = x[j]; x[j] = x[j + 1]; x[j + 1] = t; }
}
int maxFail(int *x, int n){             /* 回傳最高不及格分；無則回傳 -1 */
    int r = -1;
    for (int i = 0; i < n; i++) if (x[i] < 60 && x[i] > r) r = x[i];
    return r;
}
int minPass(int *x, int n){             /* 回傳最低及格分；無則回傳 -1 */
    int r = -1;
    for (int i = 0; i < n; i++) if (x[i] >= 60 && (r == -1 || x[i] < r)) r = x[i];
    return r;
}
int main(void){
    int n; scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);
    sortAsc(a, n);
    for (int i = 0; i < n; i++) printf("%d%c", a[i], i == n - 1 ? '\n' : ' ');

    return 0;
}

def sort_asc(x):

def max_fail(x):

def sort_count(arr):
    if len(arr) <= 1:
        return arr, 0
    mid = len(arr)//2
    left, cl = sort_count(arr[:mid])
    right, cr = sort_count(arr[mid:])
    merged, i, j, inv = [], 0, 0, cl + cr
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:

    merged.extend(left[i:]); merged.extend(right[j:])
    return merged, inv

說明（範例二）：N=8、M=3，被淘汰順序是 3、6、1、5、2、8，8 的下一位是 4，故幸運者是 4。

作法：用佇列模擬圓圈——把前 M−1 個人從隊首搬到隊尾，再把第 M 個人（隊首）淘汰，重複 K 次；最後隊首即為幸運者。

#include <stdio.h>
int q[200005];
int main(void){
    int n,m,k; scanf("%d %d %d",&n,&m,&k);
    int head=0, tail=0, sz=n;
    for(int i=1;i<=n;i++) q[tail++]=i;     /* 圓形佇列：用足夠大的線性陣列模擬 */
    for(int t=0;t<k;t++){
        for(int s=0;s<m-1;s++){            /* 前 M-1 人搬到隊尾 */
            q[tail++]=q[head++];
        }
        head++;                            /* 第 M 人（隊首）淘汰 */
        sz--;
    }
    printf("%d\n", q[head]);               /* 隊首即幸運者 */
    return 0;
}

    q.popleft()        # 淘汰
print(q[0])            # 淘汰者的下一位即現在的隊首

#include <stdio.h>
#include <string.h>
int main(void){
    char s[128];

#include <stdio.h>
#include <string.h>
int main(void){
    char a[128], b[128];
    fgets(a, sizeof(a), stdin); a[strcspn(a, "\n")] = '\0';
    fgets(b, sizeof(b), stdin); b[strcspn(b, "\n")] = '\0';
